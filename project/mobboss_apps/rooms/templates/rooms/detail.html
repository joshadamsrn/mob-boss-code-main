<script>

  (() => {

    const intervalMs = {{ room_state_poll_interval_seconds }} * 1000;

    let inFlight = false;


    const refreshRoomMembers = async () => {

      if (document.hidden || inFlight) {

        return;

      }

      inFlight = true;

      try {

        const url = new URL(window.location.href);

        url.searchParams.delete("autojoin");

        const response = await fetch(url.toString(), {

          method: "GET",

          cache: "no-store",

          credentials: "same-origin",

          headers: { "X-Requested-With": "XMLHttpRequest" }

        });

        if (response.redirected) {

          window.location.href = response.url;

          return;

        }

        if (!response.ok) {

          return;

        }


        const html = await response.text();

        const doc = new DOMParser().parseFromString(html, "text/html");

        const nextHeading = doc.getElementById("roomMembersHeading");

        const nextList = doc.getElementById("roomMembersList");

        const currentHeading = document.getElementById("roomMembersHeading");

        const currentList = document.getElementById("roomMembersList");


        if (nextHeading && currentHeading) {

          currentHeading.textContent = nextHeading.textContent;

        }

        if (nextList && currentList) {

          currentList.innerHTML = nextList.innerHTML;

        }

      } finally {

        inFlight = false;

      }

    };


    window.refreshRoomMembers = refreshRoomMembers;

    window.setInterval(refreshRoomMembers, intervalMs);

    window.addEventListener("focus", refreshRoomMembers);

  })();

<script>

  (() => {

    const shuffleUrl = "/rooms/{{ room.room_id }}/shuffle-roles";

    const csrfToken = "{{ csrf_token }}";

    const roomId = "{{ room.room_id }}";

    const intervalMs = {{ room_auto_shuffle_interval_seconds }} * 1000;

    const storageKey = `roomAutoShuffleDueAt:${roomId}`;

    const shuffleButton = document.getElementById("autoShuffleButton");

    const shuffleIcon = document.getElementById("autoShuffleIcon");

    const shuffleLabel = document.getElementById("autoShuffleLabel");

    let inFlight = false;

    let spinAnimation = null;


    const nowMs = () => Date.now();

    const readDueAt = () => Number(window.localStorage.getItem(storageKey) || "");

    const writeDueAt = (ts) => window.localStorage.setItem(storageKey, String(ts));


    let dueAtMs = readDueAt();

    if (!Number.isFinite(dueAtMs) || dueAtMs <= nowMs()) {

      dueAtMs = nowMs() + intervalMs;

      writeDueAt(dueAtMs);

    }


    const startSpin = () => {

      if (!shuffleIcon || spinAnimation) {

        return;

      }

      spinAnimation = shuffleIcon.animate(

        [{ transform: "rotate(0deg)" }, { transform: "rotate(360deg)" }],

        { duration: 700, iterations: Infinity }

      );

    };


    const stopSpin = () => {

      if (!spinAnimation) {

        return;

      }

      spinAnimation.cancel();

      spinAnimation = null;

    };


    const shuffleSilently = async () => {

      if (document.hidden || inFlight) {

        return;

      }


      inFlight = true;

      if (shuffleButton) {

        shuffleButton.disabled = true;

      }

      startSpin();

        shuffleLabel.textContent = "Shuffling...";

      }

      try {

        const response = await fetch(shuffleUrl, {

          method: "POST",

          headers: {

            "Content-Type": "application/x-www-form-urlencoded",

            "X-CSRFToken": csrfToken,

            "X-Requested-With": "XMLHttpRequest"

          },

          body: "silent=true",

          credentials: "same-origin"

        });
